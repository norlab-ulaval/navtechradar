// To build ROS packages using colcon
colcon build

// To install built packages
. install/setup.bash

// Launch radar and camera with paramaters from yaml file
ros2 launch launch_ros launch_radar_and_camera.launch.py

// Record camera data and radar data to user defined folder and file
ros2 bag record -o recordings/record_test /radar_data/configuration_data /radar_data/fft_data camera_data/image_data

// Play back example for above
ros2 bag play recordings/record_test/record_test_0.db3

// Run a tranform publisher to visulaise laser_scan data
ros2 run tf2_ros static_transform_publisher 0 0 0 0 0 0 1 map laser_frame
ros2 run tf2_ros static_transform_publisher 0 0 0 0 0 0 1 map point_cloud

// Get a parameter
ros2 param get laser_scan_publisher power_threshold
ros2 param get laser_scan_publisher azimuth_offset
ros2 param get laser_scan_publisher start_azimuth
ros2 param get laser_scan_publisher end_azimuth
ros2 param get laser_scan_publisher start_bin
ros2 param get laser_scan_publisher end_bin

// Set a parameter
ros2 param set laser_scan_publisher power_threshold 110
ros2 param set laser_scan_publisher azimuth_offset 200
ros2 param set laser_scan_publisher start_azimuth 0
ros2 param set laser_scan_publisher end_azimuth 400
ros2 param set laser_scan_publisher start_bin 0
ros2 param set laser_scan_publisher end_bin 2856

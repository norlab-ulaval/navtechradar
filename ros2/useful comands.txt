// To build ROS packages using colcon
colcon build

// To install built packages
. install/setup.bash

// Launch radar laser scan publisher with paramaters from yaml file
ros2 launch launch_laser_scan_publisher.launch.py

// Record camera data and radar data to user defined folder and file
ros2 bag record -o recordings/record_test /radar_data/configuration_data /radar_data/fft_data camera_data/image_data

// Play back example for above
ros2 bag play recordings/record_test/record_test_0.db3

// Run a tranform publisher to visulaise laser_scan data
// Note - Automatically handled if using launch script
ros2 run tf2_ros static_transform_publisher 0 0 0 0 0 0 1 map point_cloud

// Get the power_threshold parameter from the laser_scan_publisher node
ros2 param get laser_scan_publisher power_threshold

// Set the power_threshold parameter on the laser_scan_publisher node
ros2 param set laser_scan_publisher power_threshold 110

// To open RVIZ to visulaise data
// Note - Automatically handled if using launch script
rviz2
